################################################################
## USING GIT IN LINUX
################################################################
git status #shows what changes you have made to the project
git status -s #same thing but less information

git diff HEAD #shows differences between changes you've made and the last commit

git log #shows a log of commits you have made

git add file.txt #tells git to start tracking changes to this file

git remove file.txt #removes a file from the repository (doesn't delete it)

git commit -m "description of changes" #records all changes to git; any new changes can be reverted back to this commit

git commit file.txt -m 'description of changes' #same thing but for a specific file

git push origin master #pushes your commits to your online repository; master is the name of the branch

git push -u origin master #same thing, but next time you can just use git push and it will push it to origin master

git push tyler master #pushes your commits to the main DVZ repository (pending approval by me)

git checkout #resets all changes you made to the last commit

git checkout -- file.txt #same but just for file.txt

git branch name_of_branch #copies the current branch to create a new branch

git checkout name_of_branch #switches to the branch

get checkout master #switch to the master branch
git merge name_of_branch #merge name_of_branch to the master branch
git branch -d name_of_branch #deletes branch
